## Technical Description
```sql
create table conv.gdpr_requests (
  id uuid primary key default gen_random_uuid(),
  business_id uuid not null, contact_id uuid,
  kind text not null check (kind in ('erase','export','access')),
  status text not null default 'pending'
    check (status in ('pending','processing','done','failed')),
  requested_by text, result_ref text,
  created_at timestamptz not null default now(), completed_at timestamptz
);
create table core.audit_log (
  id bigint generated always as identity primary key,
  business_id uuid, actor text not null, action text not null,
  entity text, entity_id text, details jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now()
);
create or replace function conv.gdpr_erase_contact(p_contact uuid)
returns void language plpgsql security definer as $fn$
begin
  update conv.contacts set profile = '{}'::jsonb, erased_at = now()
    where id = p_contact;
  delete from conv.channel_identities where contact_id = p_contact;
  update conv.messages m set content = null
    from conv.contacts c
    where c.id = p_contact and m.conversation_id = c.conversation_id;
  update conv.orders set customer_phone = null
    where customer_phone in (select external_user_id from conv.channel_identities
                             where contact_id = p_contact);  -- rulat ÎNAINTE de delete identities? NU —
  -- ATENȚIE la ordine: orders întâi (folosește identities), identities după. Vezi pașii.
  insert into core.audit_log(actor, action, entity, entity_id)
    values ('gdpr_svc','gdpr_erase','contact', p_contact::text);
end $fn$;
```
ORDINEA CORECTĂ în funcție: (1) orders.customer_phone=NULL pe match cu identities, (2) messages.content=NULL, (3) delete identities, (4) update contacts, (5) audit. SECURITY DEFINER + grant EXECUTE doar rolului gdpr (sau owner) — bot_runtime NU o poate apela.
